class Cat{
	void greet(){
		println("MIAOMIAOMIAO");
	}
}

class Lamb{
	void greet(){
		println("MIEMIEMIE");
	}
}

class Animals{
	Cat c;
	Lamb l;
	void greet(){
		c.greet();
		l.greet();
	}
}

int main(){
	Cat cat;
	cat = new Cat;
	Lamb lamb;
	lamb = new Lamb;
	Animals a = new Animals;
	a.c = cat;
	a.l = lamb;
	a.greet();
}
